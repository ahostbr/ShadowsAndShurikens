SCRIPT_NAME = "devtools_status_dashboard.py"


def main() -> None:
    tools_root = get_tools_root()
    script = tools_root / SCRIPT_NAME
    print("[DEVTOOLS] devtools.health.devtools_status_dashboard ->", script)
    runpy.run_path(str(script), run_name="__main__")


if __name__ == "__main__":
    main()
